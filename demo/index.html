With custom request response callbacks

<script type="text/javascript">
  var editableField = 
    document.querySelector('#withCustomResponse');
  
  editableField
    .setSuccessResponseCallback(
      function(response) {
        alert("Success: " + response);
      }
    );
  
  editableField
    .setErrorResponseCallback(
      function(response, status) {
        alert("[" + status + "] " + response);
      }
    );
</script>

With validation

<script type="text/javascript">
  document.querySelector('#withValidation')
    .setValidationFunction(function(value) {
      var reg = new RegExp(/^\d+$/);
      if (!reg.test(value)) {
        alert("Value must only contain numbers!");
        return false;
      }
      return true;
    });
</script>
